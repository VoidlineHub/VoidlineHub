local dragging = false
local dragInput
local dragStart
local startPos

local function update(input)
	local delta = input.Position - dragStart
	main.Position = UDim2.new(
		startPos.X.Scale, startPos.X.Offset + delta.X,
		startPos.Y.Scale, startPos.Y.Offset + delta.Y
	)
end

main.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = main.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

main.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement then
		dragInput = input
	end
end)

UIS.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)

local function stabilize(bus)
	for _, v in ipairs(bus:GetDescendants()) do
		if v:IsA("BasePart") then
			v.AssemblyLinearVelocity = Vector3.zero
			v.AssemblyAngularVelocity = Vector3.zero
			v.CanCollide = false
			v.Anchored = true
		end
	end
end

local function unanchor(bus)
	for _, v in ipairs(bus:GetDescendants()) do
		if v:IsA("BasePart") then
			v.Anchored = false
		end
	end
end

local des = 1

local function tweenVehicle(bus, targetCF, duration)
	if not bus or not bus.PrimaryPart then return end

	Workspace.Gravity = 0
	stabilize(bus)

	local startCF = bus:GetPivot()
	local startTime = tick()
	local done = false

	currentTween = RunService.Heartbeat:Connect(function()
		if not autofarmRunning then
			currentTween:Disconnect()
			done = true
			return
		end

		local elapsed = tick() - startTime
		local remain = math.max(0, math.ceil(duration - elapsed))
		statusauto.Text = "Teleporting " .. remain .. "s"

		local alpha = math.clamp(elapsed / duration, 0, 1)
		bus:PivotTo(startCF:Lerp(targetCF, alpha))

		if alpha >= 1 then
			currentTween:Disconnect()
			done = true
		end
	end)

	repeat task.wait() until done

	bus:PivotTo(targetCF)
	stabilize(bus)
	task.wait(0.1)

	Workspace.Gravity = 196.2
	task.wait(0.2)
	unanchor(bus)

	statusauto.Text = "Arrived"
    des = des + 1
end

local function tweenVehicleHrp(bus, targetCF, duration)
	if not bus or not bus.PrimaryPart then return end

	Workspace.Gravity = 0
	stabilize(bus)

	local character = player.Character
	local hrp = character and character:FindFirstChild("HumanoidRootPart")
	local driveSeat

	for _, v in ipairs(bus:GetDescendants()) do
		if v:IsA("VehicleSeat") then
			driveSeat = v
			break
		end
	end

	local pivotCF = bus:GetPivot()
	local refCF = (hrp and hrp.CFrame) or (driveSeat and driveSeat.CFrame) or pivotCF
	local offset = pivotCF:ToObjectSpace(refCF)
	local startCF = refCF
	local startTime = tick()
	local done = false

	currentTween = RunService.Heartbeat:Connect(function()
		if not autofarmRunning then
			currentTween:Disconnect()
			done = true
			return
		end

		local elapsed = tick() - startTime
		local remain = math.max(0, math.ceil(duration - elapsed))
		statusauto.Text = "Teleporting " .. remain .. "s"

		local alpha = math.clamp(elapsed / duration, 0, 1)
		bus:PivotTo(startCF:Lerp(targetCF, alpha):ToWorldSpace(offset:Inverse()))

		if alpha >= 1 then
			currentTween:Disconnect()
			done = true
		end
	end)

	repeat task.wait() until done

	bus:PivotTo(targetCF:ToWorldSpace(offset:Inverse()))
	stabilize(bus)
	task.wait(0.1)

	Workspace.Gravity = 196.2
	task.wait(0.1)
	unanchor(bus)

	statusauto.Text = "Arrived"
end

local function getCheckpoint()
	for _, v in ipairs(checkpoints:GetDescendants()) do
		local stopGui = v:FindFirstChild("BusStop", true)
		if stopGui then
			local stopText = stopGui:FindFirstChild("Stop", true)
			if stopText and stopText:IsA("TextLabel") and stopText.Text ~= "" then
				statusauto.Text = stopText.Text
				-- warn("Next Stop:", stopText.Text)
				return v
			end
		end
	end
end

local function getBusName()
	local ignoreNames = {
		UICorner = true, UIListLayout = true,
		UIPadding = true, OwnedCarTemplate = true
	}
	for _, v in ipairs(garageUI:GetChildren()) do
		if not ignoreNames[v.Name] and v:IsA("Frame") then
			return v.Name
		end
	end
end

local function startAutofarm()
	if autofarmRunning then return end

	autofarmRunning = true
	statusauto.Text = "Preparing..."

	DespawnCar:FireServer()
	task.wait(1)

	local selectedBusName = getBusName()

	if not selectedBusName then
		statusauto.Text = "Bus Not Found"
		autofarmRunning = false
		return
	end

	-- warn("Bus:", selectedBusName)

	SpawnCar:FireServer(selectedBusName)

	local startTick = tick()

	repeat
		task.wait(0.25)
		for _, v in ipairs(spawnedVehicles:GetChildren()) do
			local owner = v:FindFirstChild("Owner", true)
			if owner and tostring(owner.Value) == player.Name then
				vehicle = v
				break
			end
		end
		if tick() - startTick > 15 then break end
	until vehicle

	if not vehicle then
		statusauto.Text = "Vehicle Not Found"
		autofarmRunning = false
		return
	end

	-- warn("Vehicle Spawned:", vehicle.Name)

	repeat task.wait()
	until vehicle.PrimaryPart or vehicle:FindFirstChildWhichIsA("BasePart")

	local driveSeat

	for _, v in ipairs(vehicle:GetDescendants()) do
		if v:IsA("VehicleSeat") then
			driveSeat = v
			break
		end
	end

	if not driveSeat then
		statusauto.Text = "Driver Seat Missing"
		autofarmRunning = false
		return
	end

	statusauto.Text = "Sitting"

	local character = player.Character or player.CharacterAdded:Wait()
	local humanoid = character:WaitForChild("Humanoid")
	local hrp = character:WaitForChild("HumanoidRootPart")

	hrp.CFrame = driveSeat.CFrame + Vector3.new(0, 3, 0)
	task.wait(1)
	driveSeat:Sit(humanoid)
	task.wait(1)

	if humanoid.SeatPart ~= driveSeat then
		hrp.CFrame = driveSeat.CFrame
		task.wait(0.5)
		driveSeat:Sit(humanoid)
	end

	statusauto.Text = "Going To Trigger"

	task.wait(2)

	local triggerPart = workspace:WaitForChild("BusJobTriggers"):WaitForChild("Baranangsiang")
	tweenVehicleHrp(vehicle, triggerPart.CFrame, 5)

	statusauto.Text = "Starting Job"
	CancelBusJob:FireServer()
	task.wait(2)

	ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("StartBusJob"):InvokeServer("Baranangsiang_ke_Cirebon2")

	statusauto.Text = "Waiting System"
	task.wait(5)

	while autofarmRunning do
		local activeCheckpoint = getCheckpoint()
        warn(tonumber(des))
        if tonumber(des) > 9 then
            local triggerPartEndJob = workspace.BusJobEndTriggers.EndBaranangsiang
	        tweenVehicleHrp(vehicle, triggerPartEndJob.CFrame, 60)
            task.wait(0)
            des = 0
            continue
        end

		if not activeCheckpoint then
			statusauto.Text = "Checkpoint Missing"
			task.wait(2)
			continue
		end

		local targetCF = activeCheckpoint:IsA("BasePart")
			and activeCheckpoint.CFrame
			or activeCheckpoint:GetPivot()

		-- ambil durasi dari textbox, fallback ke 60 kalau kosong/invalid
		local inputDuration = tonumber(durationBox.Text)
		local tweenDuration = (inputDuration and inputDuration > 0) and inputDuration or 60

		tweenVehicle(vehicle, targetCF, tweenDuration)

		task.wait(20)
	end
end
